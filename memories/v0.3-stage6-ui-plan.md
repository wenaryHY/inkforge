### 第二步：创建 API 服务层

**文件**：`src/admin/ui/src/services/mediaCategoryService.ts`

```typescript
import { apiData } from '../lib/api';
import type { MediaCategory, CreateMediaCategoryRequest, UpdateMediaCategoryRequest } from '../types';

export async function listMediaCategories(): Promise<MediaCategory[]> {
  return apiData<MediaCategory[]>('/api/admin/media/categories');
}

export async function createMediaCategory(data: CreateMediaCategoryRequest): Promise<MediaCategory> {
  return apiData<MediaCategory>('/api/admin/media/categories', {
    method: 'POST',
    body: JSON.stringify(data),
  });
}

export async function updateMediaCategory(id: string, data: UpdateMediaCategoryRequest): Promise<MediaCategory> {
  return apiData<MediaCategory>(`/api/admin/media/categories/${id}`, {
    method: 'PATCH',
    body: JSON.stringify(data),
  });
}

export async function deleteMediaCategory(id: string): Promise<void> {
  await apiData(`/api/admin/media/categories/${id}`, { method: 'DELETE' });
}
```

### 第三步：创建 Hooks

**文件**：`src/admin/ui/src/hooks/useMediaCategories.ts`

```typescript
import { useCallback, useEffect, useState } from 'react';
import * as mediaCategoryService from '../services/mediaCategoryService';
import type { MediaCategory } from '../types';

export function useMediaCategories() {
  const [categories, setCategories] = useState<MediaCategory[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const fetch = useCallback(async () => {
    setLoading(true);
    setError(null);
    try {
      const data = await mediaCategoryService.listMediaCategories();
      setCategories(data);
    } catch (err) {
      setError(err instanceof Error ? err.message : '加载分类失败');
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => {
    void fetch();
  }, [fetch]);

  return { categories, loading, error, refetch: fetch };
}
```

## 短期 vs 长期设计

### 短期实现（当前阶段）

✅ **直接集成**
- React 组件直接调用 API
- 使用 React Context 管理状态
- 内联样式 + CSS 变量

✅ **快速交付**
- 基于现有组件库快速开发
- 复用已有的 Modal、Form 等组件
- 最小化新增依赖

### 长期预留（为 Tauri/SDK 准备）

🔮 **接口设计**
- 组件接收 props，不直接调用 API
- 分离数据获取逻辑（hooks）
- 为未来的 IPC 通信预留接口

🔮 **可复用性**
- 通用组件（MediaCategoryList、MediaCategorySelect）
- 可独立使用的表单组件
- 清晰的数据流向

🔮 **扩展性**
- CSS 变量系统支持主题切换
- 组件接口支持自定义渲染
- 为多框架迁移预留 API

## 组件设计原则

### 通用组件（为 SDK 预留）

这些组件应该设计为可独立使用：

- `MediaCategorySelect` - 分类选择下拉框
- `MediaCategoryBadge` - 分类标签
- `MediaCategoryList` - 分类列表（只负责展示）

**特点**：
- 接收 props 而非直接调用 API
- 支持自定义事件处理
- 不依赖全局状态

### 管理端特定组件

这些组件是管理后台特定的：

- `MediaCategoryManager` - 分类管理页面
- `MediaCategoryForm` - 分类编辑表单

**特点**：
- 可以直接调用 API
- 使用 hooks 管理状态
- 包含管理特定的功能（删除、排序等）

## 性能考虑

### 优化策略

1. **分类缓存**
   - 分类列表变化不频繁，可缓存
   - 使用 React Query 或 SWR 考虑

2. **虚拟滚动**
   - 如果媒体数量很大，考虑虚拟滚动

3. **防抖/节流**
   - 搜索、筛选操作添加防抖

### 监控指标

- 首屏加载时间 < 2s
- API 响应时间 < 200ms
- 列表滚动帧率 > 60fps
